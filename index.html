    public static string ToCamelCase(string str)

            char firstLetter = str[0];

        str = ""; 
        if (Char.IsUpper(firstLetter))
            foreach(var e in words)
                str += e;
        else 
            for (int i = 0; i < words.Length; i++)
            { 
                temporaryString = words[i];

            }
        return str; 
    }
